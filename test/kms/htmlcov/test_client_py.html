241 def gcd(self, a, b): 

242 """ 

243 gcd 

244 """ 

245 while a != 0: 

246 a, b = b % a, a 

247 return b 

248 

249 def findModReverse(self, a, m): 

250 """ 

251 ModReverse 

252 """ 

253 if self.gcd(a, m) != 1: 

254 return None 

255 u1, u2, u3 = 1, 0, a 

256 v1, v2, v3 = 0, 1, m 

257 while v3 != 0: 

258 q = u3 // v3 

259 v1, v2, v3, u1, u2, u3 = (u1 - q * v1), (u2 - q * v2), (u3 - q * v3), v1, v2, v3 

260 return u1 % m

262 def longToBytes(self, value): 

263 result = [] 

264 i = 0 

265 while value >> (i * 8) > 0 or len(bytes(result)) % 16 != 0: 

266 result.append(int(value >> (i * 8) & 0xff)) 

267 i += 1 

268 result.reverse() 

269 result_bytes=bytes(result) 

270 return result_bytes

297 Qinv = self.longToBytes(self.findModReverse(rsa.q, rsa.p))
